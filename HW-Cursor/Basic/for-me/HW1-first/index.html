Git comands

git config --global user.name blabla = використовується для вказівки налаштувань (ім'я користувача)

git config --global user.email blabla@bla = використовується для вказівки налаштувань (електрона пошта користувача)

git init = застосовуємо для створення git репозиторія

git status = показує стан файлів в робочій директорії

git add = додає вміст робочої директорії в індекс (staging area) для подальшого коміта

git commit = зберігає зміні в новій версії файлу

git log = перегляд історії комітов, починаючи з самого свіжого та йдучи до витоків проекту

git checkout = використовуємо для переключення між комітамита гілками

git branch = редагує гілки, створює нові, видаляє гілки

git push = встановлення звя'зку з віддаленим репозиторієм, обчислення локальних змін відсутніх в ньому, і власне їх передачі в вищезгаданий репозиторій

git pull = забирає зміни із зазначеного віддаленого сховища, а потім намагається злити їх з поточною гілкою

git merge = злиття одгієї або декількох гілок в поточну

git clone = клонує репозиторій

git revert = створює антикомміт для зазначеного комміта, скасовує зміни, внесені в ньому

git reset = скасування змін

git rebase = об'єднує зміни, зроблені в одній гілці, з іншою гілкою

git cherry-pick = бере зміни, внесені будь яким коммітом, і застосує їх заново у вигляді нового комміта на горі поточної гілки

git stash = тимчасове збереження всіх незакомміченних змін для очищення робочої директорії без необхідності коммітов незавершену робот в нову гілку

gitignore = визначає навмисно неотслідковані файли які Git повинен ігнорувати
